    {
		int x= i, d= 0;
		S= 1;
		for(j= 0; j < m; j++)
		{
			if( x&1 )
            {
                S*= A[j];
                d++;
            }
			x>>= 1;
			if( S > n ) break;
		}

		{
			if( d%2 ) count+= n/S;

		}
